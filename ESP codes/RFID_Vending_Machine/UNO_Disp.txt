// UNO Display Controller — BMP Image Version
// SD card in TFT shield (pin 10) holds images: 0.bmp to 9.bmp
//
// Image mapping (put these BMP files on the SD card):
//   0.bmp  ->  IDLE        — Welcome / tap card screen
//   1.bmp  ->  AUTH        — Card accepted / authorized
//   2.bmp  ->  DENIED      — Unauthorized card
//   3.bmp  ->  VEND        — Dispensing napkin
//   4.bmp  ->  DONE        — Success / please collect
//   5.bmp  ->  EMPTY       — Out of stock
//   6.bmp  ->  LIMIT       — Daily limit reached
//   7.bmp  ->  TIMEOUT     — No selection made
//
// Commands received from ESP32 over Serial RX (pin 0):
//   IDLE            -> show 0.bmp
//   AUTH:<uid>      -> show 1.bmp
//   DENIED          -> show 2.bmp
//   VEND:1          -> show 3.bmp
//   DONE:1:<rem>    -> show 4.bmp
//   EMPTY:1         -> show 5.bmp
//   LIMIT           -> show 6.bmp
//   TIMEOUT         -> show 7.bmp

#include <MCUFRIEND_kbv.h>
#include <SD.h>
#include <SPI.h>

MCUFRIEND_kbv tft;

#define SD_CS 10  // TFT shield SD card CS pin

File bmpFile;
uint16_t colorBuffer[20];
uint8_t  sdbuffer[60];  // 20 pixels * 3 bytes each

void setup() {
  Serial.begin(115200);

  uint16_t ID = tft.readID();
  tft.begin(ID);
  tft.setRotation(0);  // Portrait
  tft.fillScreen(0x0000);

  if (!SD.begin(SD_CS)) {
    tft.setCursor(10, 10);
    tft.setTextColor(0xF800);
    tft.setTextSize(2);
    tft.print("SD Failed!");
    while (1);
  }

  drawBMP("0.bmp");  // Show idle screen on boot
}

void loop() {
  if (Serial.available()) {
    String cmd = Serial.readStringUntil('\n');
    cmd.trim();

    if (cmd == "IDLE") {
      drawBMP("0.bmp");
    } else if (cmd.startsWith("AUTH:")) {
      drawBMP("1.bmp");
    } else if (cmd == "DENIED") {
      drawBMP("2.bmp");
    } else if (cmd.startsWith("VEND:")) {
      drawBMP("3.bmp");
    } else if (cmd.startsWith("DONE:")) {
      drawBMP("4.bmp");
    } else if (cmd.startsWith("EMPTY:")) {
      drawBMP("5.bmp");
    } else if (cmd == "LIMIT") {
      drawBMP("6.bmp");
    } else if (cmd == "TIMEOUT") {
      drawBMP("7.bmp");
      delay(2000);
      drawBMP("0.bmp");  // Return to idle after timeout
    }
  }
}

// ── BMP loader ──────────────────────────────────────────────────────────────

void drawBMP(const char *filename) {
  bmpFile = SD.open(filename);
  if (!bmpFile) {
    Serial.print("Not found: ");
    Serial.println(filename);
    return;
  }

  // Validate BMP header
  if (read16(bmpFile) != 0x4D42) {
    Serial.println("Not a BMP");
    bmpFile.close();
    return;
  }

  read32(bmpFile);                        // file size (unused)
  read32(bmpFile);                        // reserved
  uint32_t bmpImageOffset = read32(bmpFile);  // pixel data offset
  read32(bmpFile);                        // DIB header size
  int bmpWidth  = read32(bmpFile);
  int bmpHeight = read32(bmpFile);

  // Validate format: 1 color plane, 24-bit, no compression
  if (read16(bmpFile) != 1 || read16(bmpFile) != 24 || read32(bmpFile) != 0) {
    Serial.println("Unsupported BMP format (need 24-bit uncompressed)");
    bmpFile.close();
    return;
  }

  // Clamp to screen size
  int drawW = min(bmpWidth,  (int)tft.width());
  int drawH = min(bmpHeight, (int)tft.height());

  uint32_t rowSize = (bmpWidth * 3 + 3) & ~3;  // rows are padded to 4 bytes

  tft.setAddrWindow(0, 0, drawW - 1, drawH - 1);
  bool first = true;

  for (int row = drawH - 1; row >= 0; row--) {
    bmpFile.seek(bmpImageOffset + (uint32_t)row * rowSize);
    int col = 0;
    while (col < drawW) {
      int batch = min(20, drawW - col);
      bmpFile.read(sdbuffer, batch * 3);
      for (int i = 0; i < batch; i++) {
        uint8_t b = sdbuffer[i * 3];
        uint8_t g = sdbuffer[i * 3 + 1];
        uint8_t r = sdbuffer[i * 3 + 2];
        colorBuffer[i] = tft.color565(r, g, b);
      }
      tft.pushColors(colorBuffer, batch, first);
      first = false;
      col += batch;
    }
  }

  bmpFile.close();
  Serial.print("OK: ");
  Serial.println(filename);
}

// ── Little-endian SD readers ─────────────────────────────────────────────────

uint16_t read16(File &f) {
  uint16_t result;
  ((uint8_t *)&result)[0] = f.read();
  ((uint8_t *)&result)[1] = f.read();
  return result;
}

uint32_t read32(File &f) {
  uint32_t result;
  ((uint8_t *)&result)[0] = f.read();
  ((uint8_t *)&result)[1] = f.read();
  ((uint8_t *)&result)[2] = f.read();
  ((uint8_t *)&result)[3] = f.read();
  return result;
}

void drawBMP(const char *filename) {
  Serial.print("Opening: ");
  Serial.println(filename);

  bmpFile = SD.open(filename);
  if (!bmpFile) {
    Serial.println("File not found");
    return;
  }

  if (read16(bmpFile) != 0x4D42) {
    Serial.println("Not a BMP file");
    bmpFile.close();
    return;
  }

  read32(bmpFile); // size
  read32(bmpFile); // reserved
  uint32_t bmpImageoffset = read32(bmpFile);
  read32(bmpFile); // DIB header size
  int bmpWidth = read32(bmpFile);
  int bmpHeight = read32(bmpFile);

  Serial.print("W: "); Serial.print(bmpWidth);
  Serial.print(" H: "); Serial.println(bmpHeight);

  if (bmpWidth > tft.width() || bmpHeight > tft.height()) {
    Serial.println("Image too large for screen.");
    bmpFile.close();
    return;
  }

  if (read16(bmpFile) != 1 || read16(bmpFile) != 24 || read32(bmpFile) != 0) {
    Serial.println("Unsupported BMP format");
    bmpFile.close();
    return;
  }

  uint32_t rowSize = (bmpWidth * 3 + 3) & ~3;

  tft.setAddrWindow(0, 0, bmpWidth - 1, bmpHeight - 1);
  bool first = true;

  for (int row = bmpHeight - 1; row >= 0; row--) {
    bmpFile.seek(bmpImageoffset + row * rowSize);
    int col = 0;
    while (col < bmpWidth) {
      int batch = min(20, bmpWidth - col);
      bmpFile.read(sdbuffer, batch * 3);
      int idx = 0;
      for (int i = 0; i < batch; i++) {
        uint8_t b = sdbuffer[idx++];
        uint8_t g = sdbuffer[idx++];
        uint8_t r = sdbuffer[idx++];
        colorBuffer[i] = tft.color565(r, g, b);
      }
      tft.pushColors(colorBuffer, batch, first);
      first = false;
      col += batch;
    }
  }

  bmpFile.close();
  Serial.println("BMP loaded.");
}

// Little-endian readers
uint16_t read16(File &f) {
  uint16_t result;
  ((uint8_t *)&result)[0] = f.read();
  ((uint8_t *)&result)[1] = f.read();
  return result;
}

uint32_t read32(File &f) {
  uint32_t result;
  ((uint8_t *)&result)[0] = f.read();
  ((uint8_t *)&result)[1] = f.read();
  ((uint8_t *)&result)[2] = f.read();
  ((uint8_t *)&result)[3] = f.read();
  return result;
}
